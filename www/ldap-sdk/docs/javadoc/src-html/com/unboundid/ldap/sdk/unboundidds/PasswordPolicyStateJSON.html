150  /**
151   * Attempts to retrieve and decode the password policy state information for
152   * the specified user.
153   *
154   * @param  connection  The connection to use to communicate with the server.
155   *                     It must not be {@code null}, and it must be established
156   *                     and authenticated as an account with permission to
157   *                     access the target user's password policy state
158   *                     information.
159   * @param  userDN      The DN of the user for whom to retrieve the password
160   *                     policy state.  It must not be {@code null}.
161   *
162   * @return  The password policy state information for the specified user, or
163   *          {@code null} because no password policy state information is
164   *          available for the user.
165   *
166   * @throws  LDAPException  If a problem is encountered while trying to
167   *                         retrieve the user's entry or decode the password
168   *                         policy state JSON object.
169   */
170  public static PasswordPolicyStateJSON get(final LDAPInterface connection,
171                                            final String userDN)
172         throws LDAPException
173  {
174    final SearchResultEntry userEntry = connection.getEntry(userDN,
175         PASSWORD_POLICY_STATE_JSON_ATTRIBUTE);
176    if (userEntry == null)
177    {
178      throw new LDAPException(ResultCode.NO_SUCH_OBJECT,
179           ERR_PW_POLICY_STATE_JSON_GET_NO_SUCH_USER.get(userDN));
180    }
181
182    return get(userEntry);
183  }

187  /**
188   * Attempts to retrieve and decode the password policy state information from
189   * the provided user entry.
190   *
191   * @param  userEntry  The entry for the user for whom to obtain the password
192   *                    policy state information.  It must not be {@code null}.
193   *
194   * @return  The password policy state information from the provided user
195   *          entry, or {@code null} if no password policy state information is
196   *          available for the user.
197   *
198   * @throws  LDAPException  If a problem is encountered while trying to decode
199   *                         the password policy state JSON object.
200   */
201  public static PasswordPolicyStateJSON get(final Entry userEntry)
202         throws LDAPException
203  {
204    final String valueString =
205         userEntry.getAttributeValue(PASSWORD_POLICY_STATE_JSON_ATTRIBUTE);
206    if (valueString == null)
207    {
208      return null;
209    }
210
211    final JSONObject jsonObject;
212    try
213    {
214      jsonObject = new JSONObject(valueString);
215    }
216    catch (final Exception e)
217    {
218      Debug.debugException(e);
219      throw new LDAPException(ResultCode.DECODING_ERROR,
220           ERR_PW_POLICY_STATE_JSON_GET_CANNOT_DECODE.get(
221                PASSWORD_POLICY_STATE_JSON_ATTRIBUTE, userEntry.getDN()),
222           e);
223    }
224
225    return new PasswordPolicyStateJSON(jsonObject);
226  }

277  /**
278   * Retrieves a list of information about any error conditions that may
279   * affect usability of the user's account.
280   *
281   * @return  A list of information about any error conditions that may affect
282   *          the usability of the user's account.  The returned list may be
283   *          empty if there are no account usability errors or if this was not
284   *          included in the password policy state JSON object.
285   */
286  public List<PasswordPolicyStateAccountUsabilityError>
287              getAccountUsabilityErrors()
288  {
289    final List<PasswordPolicyStateAccountUsabilityError> errors =
290         new ArrayList<>();
291    final List<JSONValue> values = passwordPolicyStateObject.getFieldAsArray(
292         ACCOUNT_USABILITY_ERRORS.getFieldName());
293    if (values != null)
294    {
295      for (final JSONValue v : values)
296      {
297        if (v instanceof JSONObject)
298        {
299          final JSONObject o = (JSONObject) v;
300          final String typeName = o.getFieldAsString(USABILITY_FIELD_TYPE_NAME);
301          final Integer typeID = o.getFieldAsInteger(USABILITY_FIELD_TYPE_ID);
302          final String message = o.getFieldAsString(USABILITY_FIELD_MESSAGE);
303          if ((typeName != null) && (typeID != null))
304          {
305            errors.add(new PasswordPolicyStateAccountUsabilityError(typeID,
306                 typeName, message));
307          }
308        }
309      }
310    }
311
312    return Collections.unmodifiableList(errors);
313  }
